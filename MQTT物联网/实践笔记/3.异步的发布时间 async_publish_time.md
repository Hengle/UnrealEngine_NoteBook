## MQTT实践
它是一个精心设计的，MQTT数据监视器和发布者的有用示例。使用c++异步客户端接口。
MQTT应用程序的一种非常常见的用法是监视传感器，并在读数发生“重大”变化时发布读数(无论变化是什么)。
==这可能是温度，压力，湿度，土壤湿度，二氧化碳水平，或类似的东西。== 
由于我们在这个例子中没有通用的传感器，所以我们只使用时间本身作为输出数据。我们定期“采样”时间值，当它的变化超过我们所需的增量时，我们发布时间。在本例中，我们使用系统时钟，以毫秒精度测量时间。

##### 示例说明
- 连接到一个MQTT服务或者broker
- 示例的一个值
- 发布消息 使用一个“主题”对象（Topic）
- 最终或者最后遗嘱
- 通过lambdas 来回调
- 使用“创建选项”
- 使用构建类创建选项
- 在客户端的离线缓冲


```

#include <iostream>	//输入输出
#include <cstdlib>	//C的Std库
#include <string>	// 字符串
#include <thread>	// For sleep
#include <atomic>	// 原子操作
#include <chrono>	// 时间操作
#include <cstring>	// C类型字符串
#include "mqtt/async_client.h"	//mqtt异步客户端
 // 输入输出
using namespace std;
// 时间操作
using namespace std::chrono;		

// DFLT服务端地址
const std::string DFLT_SERVER_ADDRESS { "tcp://localhost:1883" };

// The QoS for sending data
// 发送消息的服务质量
const int QOS = 1;

// How often to sample the "data"
// 采样频率（5ms）
const auto SAMPLE_PERIOD = milliseconds(5);

// How much the "data" needs to change before we publish a new value.
// 多少数据需要更改 在我们发布一新值之前
const int DELTA_MS = 100;

// How many to buffer while off-line
// 当离线拥有多少的缓冲
const int MAX_BUFFERED_MESSAGES = 1200;

// --------------------------------------------------------------------------
// Gets the current time as the number of milliseconds since the epoch:
// 获取当前时间作为自epoch以来的毫秒数:
// like a time_t with ms resolution.
// 像一个 time_t 是毫米的结果

//时间戳
uint64_t timestamp()
{
	auto now = system_clock::now();					//当前系统时间
	auto tse = now.time_since_epoch();				//since 时期时间 
	auto msTm = duration_cast<milliseconds>(tse);	//持续时间花费
	return uint64_t(msTm.count());					//返回持续时间的值
}

// --------------------------------------------------------------------------

int main(int argc, char* argv[])
{
	// The server URI (address)
	// 服务地址URI（地址）
	string address = (argc > 1) ? string(argv[1]) : DFLT_SERVER_ADDRESS;

	// The amount of time to run (in ms). Zero means "run forever".
	// （在ms内）运行的次数，0代表“永远运行”
	uint64_t trun = (argc > 2) ? stoll(argv[2]) : 0LL;
	// 输出： 初始化服务的地址
	cout << "Initializing for server '" << address << "'..." << endl;

	// We configure to allow publishing to the client while off-line,
	// 我们配置允许“发送”到客户端 当离线的时候
	// and that it's OK to do so before the 1st successful connection.
	// 并且 在第一次成功连接之前，它是OK的
	// 创建选项 = mqtt::创建选项构建器.当断连的时候发送（真，真），消息的最大缓冲（消息最大缓冲），删除最老的消息，结束
	auto createOpts = mqtt::create_options_builder()
						  .send_while_disconnected(true, true)
					      .max_buffered_messages(MAX_BUFFERED_MESSAGES)
						  .delete_oldest_messages()
						  .finalize();
	//mqtt的异步客户端 客户端（地址，“”，创建选项）；
	mqtt::async_client cli(address, "", createOpts);

	// 当连接并且连接丢失时设置回调函数 
	// Set callbacks for when connected and connection lost.

	//客户端.设置连接句柄（客户端（字符串））
	cli.set_connected_handler([&cli](const std::string&) {
		//输出: 连接“时间戳”
		std::cout << "*** Connected ("
			<< timestamp() << ") ***" << std::endl;
	});
	//客户端.设置连接丢失的句柄
	cli.set_connection_lost_handler([&cli](const std::string&) {
		//输出：连接丢失“时间戳”
		std::cout << "*** Connection Lost ("
			<< timestamp() << ") ***" << std::endl;
	});

	//将要发送的消息 = mqtt::消息（“测试事件”，“发布者断连的时间”，1，true）
	auto willMsg = mqtt::message("test/events", "Time publisher disconnected", 1, true);
	// 连接选型 = mqtt：连接选项构建器
	auto connOpts = mqtt::connect_options_builder()
		.clean_session()									//清理会话
		.will(willMsg)										//将要（将要发送的消息）
		.automatic_reconnect(seconds(1), seconds(10))		//自动重连（1~10秒）
		.finalize();										//终结

	try {
		// Note that we start the connection, but don't wait for completion.
		// We configured to allow publishing before a successful connection.
		// 提示： 我们开始连接，但是 不想完成
		// 在一个成功连接之前，我们配置允许发布
		cout << "Starting connection..." << endl;
		// 客户端连接
		cli.connect(connOpts);
		// 自动类型 主题 = mqtt::主题（客户端,"数据、商家"，服务质量为1）
		auto top = mqtt::topic(cli, "data/time", QOS);
		// 输出： 发布数据
		cout << "Publishing data..." << endl;
		// 当时间戳%延时毫秒不等于 = 0 ，就卡在这
		while (timestamp() % DELTA_MS != 0)
			;
		// t 时间戳
		uint64_t	t = timestamp(),
					//最后时间
					tlast = t,
					//开始时间
					tstart = t;
		// 主题发布（t）
		top.publish(to_string(t));
		// 循环
		while (true) {
			//线程休眠
			this_thread::sleep_for(SAMPLE_PERIOD);
			//时间戳
			t = timestamp();
			//cout << t << endl;
			//当前时间减去 最后时间 大于 延时时
			if (abs(int(t - tlast)) >= DELTA_MS)
				//主题 发布（字符串t (同步操作刷新tlast)）
				top.publish(to_string(tlast = t));
			// 运行次数》0 t >= (运行次数+启动时间)
			if (trun > 0 && t >= (trun + tstart))
				// 跳出循环
				break;
		}
		// 断开连接
		// Disconnect
		// 输出：断开连接
		cout << "\nDisconnecting..." << endl;
		// 客户端.断开连接-等待（0
		cli.disconnect()->wait();
		// 输出：Ok
		cout << "  ...OK" << endl;
	}
	// 报错的话 输出报错信息
	catch (const mqtt::exception& exc) {
		cerr << exc.what() << endl;
		return 1;
	}

 	return 0;
}
```
